import requests import shutil def download_random_photo(url, filename): response = requests.get(url, stream=True) if response.status_code == 200: with open(filename, 'wb') as f: response.raw.decode_content = True shutil.copyfileobj(response.raw, f) print("Photo downloaded successfully.") else: print("Failed to download photo.") def main(): url = "https://source.unsplash.com/random/" filename = "random_photo.jpg" download_random_photo(url, filename) if __name__ == "__main__": main()
